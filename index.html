const express = require('express'); const app = express(); const PORT = 3000; // JSON 데이터 파싱을 위한 미들웨어 app.use(express.json()); // [임시 데이터베이스] 서버가 켜져 있는 동안 메모리에 저장됩니다. let users = { "student_01": { id: "student_01", name: "김철수", trustScore: 60.0 }, "student_02": { id: "student_02", name: "이영희", trustScore: 80.0 } }; let reviews = []; /** * 신뢰도 지수(EMA) 계산 함수 */ function calculateNewTrustScore(oldScore, currentProjectReviews, hasDroppedOut = false) { const ALPHA = 0.2; const DEFAULT_SCORE = 50.0; const currentOldScore = oldScore ?? DEFAULT_SCORE; if (!currentProjectReviews || currentProjectReviews.length === 0) { return currentOldScore; } // 1. 이번 프로젝트에서 받은 모든 점수 합산 let totalScoreSum = 0; currentProjectReviews.forEach(review => { totalScoreSum += (review.punctuality + review.communication + review.contribution); }); // 2. 100점 만점 기준으로 환산 (항목 3개 * 5점 만점 = 평가자당 15점 만점) const maxPossibleScore = currentProjectReviews.length * 15; const currentProjectAverage100 = (totalScoreSum / maxPossibleScore) * 100; // 3. 중도 탈퇴 페널티 적용 const penaltyWeight = hasDroppedOut ? 0.8 : 1.0; // 4. PRD 공식 적용 let newScore = currentOldScore * (1 - ALPHA) + ALPHA * currentProjectAverage100 * penaltyWeight; // 5. 0~100점 제한 및 반올림 newScore = Math.max(0, Math.min(100, newScore)); return Math.round(newScore * 10) / 10; } /** * [API 엔드포인트] 피어 리뷰 제출 및 신뢰도 실시간 반영 */ app.post('/api/projects/:projectId/review', (req, res) => { const { targetUserId, punctuality, communication, contribution, comment, hasDroppedOut } = req.body; const projectId = req.params.projectId; // 1. 필수 데이터 입력 검증 if (!targetUserId || punctuality === undefined || communication === undefined || contribution === undefined) { return res.status(400).json({ success: false, message: "모든 정량 평가 항목(일정, 소통, 기여도)을 입력해주세요." }); } // 2. 정성 평가 글자수 제한 검증 (20자 이상) if (!comment || comment.trim().length < 20) { return res.status(400).json({ success: false, message: `정성 평가는 최소 20자 이상 필수입니다. (현재 ${comment ? comment.trim().length : 0}자)` }); } // 3. 존재하는 유저인지 검증 const targetUser = users[targetUserId]; if (!targetUser) { return res.status(404).json({ success: false, message: "존재하지 않는 대학생 회원입니다." }); } // 4. 새 리뷰 데이터를 배열에 저장 (DB 삽입 대체) const newReview = { projectId, targetUserId, punctuality: Number(punctuality), communication: Number(communication), contribution: Number(contribution), comment: comment.trim(), createdAt: new Date() }; reviews.push(newReview); // 5. 해당 유저가 '이번 프로젝트'에서 받은 모든 리뷰 필터링 const currentProjectReviews = reviews.filter(r => r.targetUserId === targetUserId && r.projectId === projectId); // 6. 알고리즘 기반 새 신뢰도 점수 산출 const updatedScore = calculateNewTrustScore(targetUser.trustScore, currentProjectReviews, !!hasDroppedOut); // 7. 유저 정보에 새로운 점수 반영 users[targetUserId].trustScore = updatedScore; // 8. 성공 응답 반환 return res.status(200).json({ success: true, message: `${targetUser.name}님에 대한 리뷰가 반영되어 신뢰도 지수가 업데이트되었습니다.`, data: { userId: targetUserId, userName: targetUser.name, previousScore: targetUser.trustScore, newTrustScore: updatedScore } }); }); // 서버 시작 app.listen(PORT, () => { console.log(`🚀 CamPoly 백엔드가 http://localhost:${PORT} 에서 정상 작동 중입니다!`); });
