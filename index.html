// server.js (또는 routes/review.js) const express = require('express'); const router = express.Router(); /** * [문제 분해] 신뢰도 지수(EMA) 계산 함수 * @param {number} oldScore - 기존 신뢰도 점수 (기본값 50) * @param {Array} ratings - 이번 프로젝트에서 팀원들에게 받은 평점 배열 * @param {boolean} hasDroppedOut - 중도 탈퇴 여부 (페널티 적용용) */ function calculateNewTrustScore(oldScore, ratings, hasDroppedOut = false) { const ALPHA = 0.2; // 최근 프로젝트 반영 가중치 (α) const DEFAULT_SCORE = 50; const currentOldScore = oldScore ?? DEFAULT_SCORE; if (!ratings || ratings.length === 0) return currentOldScore; // 1. 받은 점수 총합 계산 (일정준수, 소통, 기여도 각 5점 만점) let totalScoreSum = 0; ratings.forEach(review => { totalScoreSum += (review.punctuality + review.communication + review.contribution); }); // 2. 100점 만점 기준으로 환산 (최대 가능 점수 = 평가자 수 * 15점) const maxPossibleScore = ratings.length * 15; const currentProjectAverage100 = (totalScoreSum / maxPossibleScore) * 100; // 3. 중도 탈퇴 페널티 가중치 설정 (W_penalty) const penaltyWeight = hasDroppedOut ? 0.8 : 1.0; // 4. PRD 공식 적용: T_new = T_old * (1 - α) + α * (이번평균) * W_penalty let newScore = currentOldScore * (1 - ALPHA) + ALPHA * currentProjectAverage100 * penaltyWeight; // 점수 범위 제한 (0점 ~ 100점) 및 소수점 첫째짜리 반올림 newScore = Math.max(0, Math.min(100, newScore)); return Math.round(newScore * 10) / 10; } // [API 엔드포인트] 피어 리뷰 제출 및 신뢰도 업데이트 router.post('/api/projects/:projectId/review', async (req, res) => { try { const { targetUserId, punctuality, communication, contribution, comment } = req.body; // 1. [유효성 검사] 필수 데이터 확인 및 정성 평가 글자수(20자) 제한 분해 if (!targetUserId || !punctuality || !communication || !contribution) { return res.status(400).json({ success: false, message: "모든 평가 항목을 채워주세요." }); } if (!comment || comment.length < 20) { return res.status(400).json({ success: false, message: "정성 평가는 최소 20자 이상 작성해야 합니다." }); } // 2. [DB 작업 - 가상 코드] 리뷰 데이터 저장 // await db.Review.create({ targetUserId, punctuality, communication, contribution, comment }); // 3. [DB 작업 - 가상 코드] 피평가자의 기존 점수 및 이번 프로젝트에서 받은 모든 리뷰 조회 // const user = await db.User.findById(targetUserId); // const allReviewsInThisProject = await db.Review.find({ targetUserId, projectId: req.params.projectId }); // 테스트를 위한 가상 데이터 (기존 점수 60점이라 가정) const mockOldScore = 60; const mockAllReviews = [ { punctuality, communication, contribution }, // 현재 방금 제출된 리뷰 { punctuality: 4, communication: 5, contribution: 4 } // 다른 팀원이 기존에 냈던 리뷰 예시 ]; // 4. 새로운 신뢰도 지수 계산 분해 const updatedScore = calculateNewTrustScore(mockOldScore, mockAllReviews, false); // 5. [DB 작업 - 가상 코드] 유저의 새로운 신뢰도 점수 업데이트 // await db.User.updateById(targetUserId, { trustScore: updatedScore }); return res.status(200).json({ success: true, message: "리뷰가 성공적으로 반영되었습니다.", data: { newTrustScore: updatedScore } }); } catch (error) { console.error(error); return res.status(500).json({ success: false, message: "서버 오류가 발생했습니다." }); } }); module.exports = router;
